generateVolumeBoxes(orderBlock block, float orderVolume) =>
    volumeArray = array.new_float(amountOfBoxes, 0)
    boxesArray = array.new_box()
    baseVolume = ta.sma(volume, 20)
    strengthFactor = baseVolume > 0 ? math.max(0.45, math.min(1.0, 0.45 + (orderVolume / baseVolume) / 3.0)) : 0.70

    for i = 0 to amountOfBoxes - 1
        x = amountOfBoxes > 1 ? i / (amountOfBoxes - 1.0) : 0.5
        peakA = math.exp(-math.pow((x - 0.38) / 0.22, 2))
        peakB = math.exp(-math.pow((x - 0.72) / 0.16, 2)) * 0.58
        ridge = math.max(0.08, math.min(1.0, (peakA + peakB + 0.10 * math.sin(i * 1.7)) * strengthFactor))
        array.set(volumeArray, i, ridge)

        shade = int(math.round(78 - ridge * 50))
        accentColor = block.direction == "Bull" ? color.new(#00c2a8, shade) : color.new(#ff3b58, shade)

        mountainBox = box.new(
             left=time,
             top=block.topValue,
             right=time,
             bottom=block.botValue,
             border_color=color.new(accentColor, 100),
             border_width=0,
             xloc=xloc.bar_time,
             bgcolor=accentColor,
             extend=extend.none)

        array.push(boxesArray, mountainBox)

    [boxesArray, volumeArray, orderVolume]

updateBoxes(orderBlock block, currentTime) =>
    if array.size(block.boxArray) > 0
        timeLength = currentTime - block.leftTime
        railMaxWidth = math.max(1, math.round(timeLength * railWidth))
        zoneHeight = block.topValue - block.botValue
        columnCount = array.size(block.boxArray)
        columnWidth = math.max(1, math.round(railMaxWidth / columnCount))
        mountainLeft = currentTime - railMaxWidth

        for i = 0 to columnCount - 1
            ridge = i < array.size(block.boxVolumes) ? array.get(block.boxVolumes, i) : 0.20
            peakHeight = zoneHeight * mountainHeight * ridge
            selectedBox = array.get(block.boxArray, i)
            leftX = mountainLeft + columnWidth * i
            rightX = i == columnCount - 1 ? currentTime : leftX + columnWidth

            box.set_left(selectedBox, leftX)
            box.set_right(selectedBox, rightX)

            if block.direction == "Bull"
                box.set_top(selectedBox, block.botValue + peakHeight)
                box.set_bottom(selectedBox, block.botValue)
            else
                box.set_top(selectedBox, block.topValue)
                box.set_bottom(selectedBox, block.topValue - peakHeight)

    if not na(block.topLine) and not na(block.botLine)
        line.set_x2(block.topLine, currentTime)
        line.set_x2(block.botLine, currentTime)

    if not na(block.volumeLabel) and showVolume
        blockCenter = (block.topValue + block.botValue) / 2
        strength = ta.sma(volume, 20) > 0 ? block.totalVolume / ta.sma(volume, 20) : na
        badgeText = (block.direction == "Bull" ? "BUY OB" : "SELL OB") + "\nVol " + formatVolume(block.totalVolume) + "\nStr " + (na(strength) ? "-" : str.tostring(strength, "#.##") + "x")
        badgeColor = block.direction == "Bull" ? color.new(#006b5c, 12) : color.new(#8f1d2e, 12)
        label.set_x(block.volumeLabel, currentTime)
        label.set_y(block.volumeLabel, blockCenter)
        label.set_text(block.volumeLabel, badgeText)
        label.set_color(block.volumeLabel, badgeColor)
        label.set_textcolor(block.volumeLabel, volumeTextColor)
        label.set_style(block.volumeLabel, label.style_label_left)

wipeBlock(orderBlock block) =>
    if not na(block.topLine)
        line.delete(block.topLine)
    if not na(block.botLine)
        line.delete(block.botLine)
    if not na(block.bgFill)
        linefill.delete(block.bgFill)

    if not na(block.volumeLabel)
        label.delete(block.volumeLabel)

    if not na(block.volumeLine)
        line.delete(block.volumeLine)

    if array.size(block.boxArray) > 0
        for i = array.size(block.boxArray) - 1 to 0
            selectedBox = array.get(block.boxArray, i)
            box.delete(selectedBox)

checkObCondition(tuning) =>
    bear = false
    for i = tuning - 1 to 0
        start = tuning - 1
        if i == start
            if close[i] <= open[i]
                break
        else
            if close[i] > open[i]
                break

            if i == 0
                bear := true

    bull = false
    for i = tuning - 1 to 0
        start = tuning - 1
        if i == start
            if close[i] >= open[i]
                break
        else
            if close[i] < open[i]
                break

            if i == 0
                bull := true

    bullAlt = (high[tuning] - low[tuning]) > atr[tuning] * 0.8 and close[tuning] > open[tuning] and low[tuning] < low[tuning + 1]
    bearAlt = (high[tuning] - low[tuning]) > atr[tuning] * 0.8 and close[tuning] < open[tuning] and high[tuning] > high[tuning + 1]

    [bear or bearAlt, bull or bullAlt]

float obResultR = na

if trade_on and obTradeDir == 1 and bar_index > obEntryBar
    obTPHit := high >= obTP
    obSLHit := low <= obSL
    if obTPHit or obSLHit
        obExitNow := true
        obExitPrice := obTPHit and not obSLHit ? obTP : obSL
        risk = math.abs(obEntry - obSL)
        obResultR := risk > 0 ? (obExitPrice - obEntry) / risk : 0.0

if trade_on and obTradeDir == -1 and bar_index > obEntryBar
    obTPHit := low <= obTP
    obSLHit := high >= obSL
    if obTPHit or obSLHit
        obExitNow := true
        obExitPrice := obTPHit and not obSLHit ? obTP : obSL
        risk = math.abs(obEntry - obSL)
        obResultR := risk > 0 ? (obEntry - obExitPrice) / risk : 0.0

if obExitNow and not na(obResultR)
    barsHeld = bar_index - obEntryBar
    msHeld = time - obEntryTime

    obTotalTrades += 1
    obNetR += obResultR
    obTotalBarsHeld += barsHeld
    obTotalMsHeld += msHeld

    obBestR := na(obBestR) ? obResultR : math.max(obBestR, obResultR)
    obWorstR := na(obWorstR) ? obResultR : math.min(obWorstR, obResultR)

    if obResultR > 0
        obWins += 1
        obSumWinR += obResultR
        obTPStreak += 1
        obSLStreak := 0
        obBestTPStreak := math.max(obBestTPStreak, obTPStreak)

        if obTradeDir == 1
            obBuyWins += 1
        else
            obSellWins += 1
    else
        obLosses += 1
        obSumLossR += math.abs(obResultR)
        obSLStreak += 1
        obTPStreak := 0
        obWorstSLStreak := math.max(obWorstSLStreak, obSLStreak)

    if trade_showVisuals
        if trade_keepClosedVisuals
            if not na(obEntryLine)
                line.set_x2(obEntryLine, bar_index)
            if not na(obSLLine)
                line.set_x2(obSLLine, bar_index)
            if not na(obTPLine)
                line.set_x2(obTPLine, bar_index)

            exitText = obResultR > 0 ? "TP" : "SL"
            obExitLabel := label.new(
                 bar_index,
                 obExitPrice,
                 exitText + "\n" + str.tostring(obResultR, "#.##") + "R\n" + str.tostring(barsHeld) + " bars",
                 style=obResultR > 0 ? label.style_label_up : label.style_label_down,
                 color=color.new(color.black, 0),
                 textcolor=obResultR > 0 ? color.lime : color.red,
                 size=size.small)
        else
            if not na(obEntryLine)
                line.delete(obEntryLine)
            if not na(obSLLine)
                line.delete(obSLLine)
            if not na(obTPLine)
                line.delete(obTPLine)
            if not na(obEntryLabel)
                label.delete(obEntryLabel)

    obTradeDir := 0
    obEntry := na
    obSL := na
    obTP := na
    obEntryBar := na
    obEntryTime := na
    obEntryStrength := na
    obEntryVolume := na
    obEntryBlockType := "None"
    obEntryLine := na
    obSLLine := na
    obTPLine := na
    obEntryLabel := na

liveBarsHeld = obTradeDir != 0 and not na(obEntryBar) ? bar_index - obEntryBar : na
liveMsHeld = obTradeDir != 0 and not na(obEntryTime) ? time - obEntryTime : na

var table obStatsTable = na

if showTradeStats and barstate.islast and na(obStatsTable)
    obStatsTable := table.new(f_pos(statsPosition), 2, 17, bgcolor=tradeTableBg, border_color=tradeBorderCol, border_width=1, frame_color=tradeBorderCol, frame_width=1)

if showTradeStats and barstate.islast and not na(obStatsTable)
    winRate = obTotalTrades > 0 ? obWins * 100.0 / obTotalTrades : na
    profitFactor = obSumLossR > 0 ? obSumWinR / obSumLossR : na
    avgR = obTotalTrades > 0 ? obNetR / obTotalTrades : na
    avgBars = obTotalTrades > 0 ? obTotalBarsHeld / obTotalTrades : na
    avgMs = obTotalTrades > 0 ? obTotalMsHeld / obTotalTrades : na
    buyWR = obBuyTrades > 0 ? obBuyWins * 100.0 / obBuyTrades : na
    sellWR = obSellTrades > 0 ? obSellWins * 100.0 / obSellTrades : na

    statusText = obTradeDir == 1 ? "BUY ACTIVE" : obTradeDir == -1 ? "SELL ACTIVE" : "FLAT"
    statusColor = obTradeDir == 1 ? tradeBuyCol : obTradeDir == -1 ? tradeSellCol : tradeMuted
    liveRisk = obTradeDir != 0 ? math.abs(obEntry - obSL) : na
    liveR = obTradeDir == 1 and not na(liveRisk) and liveRisk > 0 ? (close - obEntry) / liveRisk : obTradeDir == -1 and not na(liveRisk) and liveRisk > 0 ? (obEntry - close) / liveRisk : na
    openTradeText = obTradeDir == 1 ? "BUY waiting TP/SL" : obTradeDir == -1 ? "SELL waiting TP/SL" : "No active trade"

    table.clear(obStatsTable, 0, 0, 1, 16)

    f_tradeCell(obStatsTable, 0, 0, "ACTIVE OB TRADE", tradeTitleCol, tradeHeaderBg)
    table.merge_cells(obStatsTable, 0, 0, 1, 0)

    f_tradeCell(obStatsTable, 0, 1, "Status", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 1, statusText, statusColor, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 2, "Open Trade", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 2, openTradeText, statusColor, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 3, "Block", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 3, obEntryBlockType, tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 4, "Entry / SL / TP", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 4, obTradeDir != 0 ? str.tostring(obEntry, format.mintick) + " / " + str.tostring(obSL, format.mintick) + " / " + str.tostring(obTP, format.mintick) : "-", tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 5, "Live R", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 5, na(liveR) ? "-" : str.tostring(liveR, "#.##") + "R", na(liveR) or liveR >= 0 ? color.lime : color.red, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 6, "Timer", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 6, obTradeDir != 0 ? str.tostring(liveBarsHeld) + " bars / " + f_msToClock(liveMsHeld) : "-", tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 7, "Strength / Volume", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 7, na(obEntryStrength) ? "-" : str.tostring(obEntryStrength, "#.##") + "x / " + formatVolume(obEntryVolume), tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 8, "PERFORMANCE", tradeTitleCol, tradeHeaderBg)
    table.merge_cells(obStatsTable, 0, 8, 1, 8)

    f_tradeCell(obStatsTable, 0, 9, "Trades", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 9, str.tostring(obTotalTrades), tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 10, "Wins / Losses", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 10, str.tostring(obWins) + " / " + str.tostring(obLosses), tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 11, "Win Rate", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 11, na(winRate) ? "-" : str.tostring(winRate, "#.##") + "%", tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 12, "Net R / Avg R", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 12, str.tostring(obNetR, "#.##") + " / " + (na(avgR) ? "-" : str.tostring(avgR, "#.##")), obNetR >= 0 ? color.lime : color.red, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 13, "Profit Factor", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 13, na(profitFactor) ? "-" : str.tostring(profitFactor, "#.##"), tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 14, "Best / Worst R", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 14, (na(obBestR) ? "-" : str.tostring(obBestR, "#.##")) + " / " + (na(obWorstR) ? "-" : str.tostring(obWorstR, "#.##")), tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 15, "Avg Hold", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 15, na(avgBars) ? "-" : str.tostring(avgBars, "#") + " bars / " + f_msToClock(avgMs), tradeTxt, tradeTableBg)

    f_tradeCell(obStatsTable, 0, 16, "Buy WR / Sell WR", tradeMuted, tradeTableBg)
    f_tradeCell(obStatsTable, 1, 16, (na(buyWR) ? "-" : str.tostring(buyWR, "#.#") + "%") + " / " + (na(sellWR) ? "-" : str.tostring(sellWR, "#.#") + "%"), tradeTxt, tradeTableBg)
